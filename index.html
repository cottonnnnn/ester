from telegram import InlineKeyboardButton, InlineKeyboardMarkup from telegram.ext import Updater, CommandHandler, CallbackQueryHandler # Функция для старта, которая отправляет меню def start(update, context): keyboard = [ [InlineKeyboardButton("Telegram Premium", callback_data='premium')], [InlineKeyboardButton("Подарок", callback_data='gift')], [InlineKeyboardButton("Звезды", callback_data='stars')] ] reply_markup = InlineKeyboardMarkup(keyboard) update.message.reply_text('Привет! Выбери что-то из меню:', reply_markup=reply_markup) # Функция для обработки нажатия кнопки def button(update, context): query = update.callback_query query.answer() if query.data == 'premium': query.edit_message_text(text="Ты выбрал Telegram Premium! Сюрприз для тебя...") elif query.data == 'gift': query.edit_message_text(text="Ты выбрал Подарок! Ожидай сюрприз...") elif query.data == 'stars': query.edit_message_text(text="Ты выбрал Звезды! Готовься к звездами...") else: query.edit_message_text(text="Что-то пошло не так...") def main(): # Токен твоего бота updater = Updater("YOUR_BOT_TOKEN", use_context=True) # Диспетчер для добавления обработчиков dp = updater.dispatcher # Команда start dp.add_handler(CommandHandler("start", start)) # Обработчик для нажатий кнопок dp.add_handler(CallbackQueryHandler(button)) # Запуск бота updater.start_polling() updater.idle() if name == '__main__': main()
